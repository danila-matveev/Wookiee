       ```python
       async def _run_chain_with_retry(orchestrator, task, task_type, context, max_retries=2):
           for attempt in range(max_retries + 1):
               result = await orchestrator.run_chain(task=task, task_type=task_type, context=context)
               if _is_substantial(result):
                   return result
               if attempt < max_retries:
                   logger.warning(f"Empty chain result (attempt {attempt+1}/{max_retries}), retrying")
           return None

       def _is_substantial(result: ChainResult) -> bool:
           detailed = result.detailed or ""
           if len(detailed) < 200:
               return False
           if "##" not in detailed:
               return False
           return True
       ```

    4. **Validate sections + graceful degradation** (per D-09, D-10, D-11, REL-03, REL-04, REL-05):
       ```python
       def _load_required_sections(report_type: ReportType) -> list[str]:
           """Parse ## headings from template file. If template not found, return empty list."""
           config = REPORT_CONFIGS[report_type]
           if not config.template_path:
               return []
           try:
               with open(config.template_path) as f:
                   content = f.read()
               return [line.strip() for line in content.splitlines()
                       if line.strip().startswith("## ")]
           except FileNotFoundError:
               logger.warning(f"Template not found: {config.template_path}")
               return []

       DEGRADATION_PLACEHOLDER = (
           "Данные для этой секции временно недоступны. "
           "Агент не смог получить информацию из источника данных. "
           "Рекомендуется проверить подключение к источникам данных "
           "и повторить генерацию отчёта."
       )

       def validate_and_degrade(report_md: str, report_type: ReportType) -> str:
           required = _load_required_sections(report_type)
           for heading in required:
               if heading not in report_md:
                   report_md += f"\n\n{heading}\n\n{DEGRADATION_PLACEHOLDER}\n"
           return report_md

       def has_substantial_content(report_md: str, report_type: ReportType) -> bool:
           """At least one section has real content (not just placeholder)."""
           required = _load_required_sections(report_type)
           if not required:
               # No template → can't validate sections, check overall length
               return len(report_md) > 500
           real_count = 0
           for heading in required:
               idx = report_md.find(heading)
               if idx >= 0:
                   # Get content between this heading and next heading or end
                   next_heading_idx = len(report_md)
                   for other in required:
                       other_idx = report_md.find(other, idx + len(heading))
                       if other_idx > idx and other_idx < next_heading_idx:
                           next_heading_idx = other_idx
                   section_content = report_md[idx + len(heading):next_heading_idx].strip()
                   if section_content and DEGRADATION_PLACEHOLDER not in section_content:
                       real_count += 1
           return real_count > 0
       ```

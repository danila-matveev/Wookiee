         - "daily": (target_date.isoformat(), target_date.isoformat())

         Default now = datetime.now(). Return now.hour in STUB_HOURS and now.minute < 35.

         Default now = datetime.now(). Return now.hour >= 17 and now.minute >= 55.

       ```python
       async def run_schedule(target_date, llm, model, pricing, notion, gate_checker, alerter):
           types = get_types_for_today(target_date)
           if not types:
               logger.info("No report types scheduled for today")
               return

           now = datetime.now()
           published_any = False

           for rt in types:
               if is_locked(rt.value, target_date):
                   logger.info(f"Skipping {rt.value} — already published today")
                   continue

               orchestrator = build_orchestrator(rt.value, llm, model, pricing)
               config = REPORT_CONFIGS[rt]
               date_from, date_to = compute_date_range(config.period, target_date)

               result = await run_report(
                   report_type=rt, target_date=target_date,
                   orchestrator=orchestrator, notion_client=notion,
                   alerter=alerter, gate_checker=gate_checker,
                   date_from=date_from, date_to=date_to,
               )

               if result.success:
                   acquire_lock(rt.value, target_date)
                   published_any = True
                   logger.info(f"Published {rt.value}: {result.notion_url}")
                   # D-14: Telegram message with type name + metrics + Notion link
                   # is handled INSIDE report_pipeline.py Step 7 (chain_result.telegram_summary + notion_url).
                   # Runner does NOT send its own Telegram for successful reports.
               elif result.skipped:
                   logger.info(f"Skipped {rt.value}: {result.reason}")
               else:
                   logger.error(f"Failed {rt.value}: {result.reason}")

           # Stub/final notifications (D-07, D-08)
           if not published_any and not any_lock_today(target_date):
               if is_final_window(now):
                   await alerter.send_alert(
                       "Данные не появились за день. Ни один отчёт не был опубликован."
                   )
               elif should_send_stub(now):
                   await alerter.send_alert(
                       "Данные пока не готовы, отслеживаем. "
                       f"Запланировано типов: {len(types)}."
                   )
       ```

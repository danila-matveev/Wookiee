function syncStatus(targetSheet, surname, status, targetCol) {
  const data = targetSheet.getDataRange().getValues();
  for (let i = 0; i < data.length; i++) {
    let targetSurnameCol = 1; // По умолчанию фамилия в первой колонке
    if (targetSheet.getName() === 'База данных креаторов') {
      targetSurnameCol = 3; // Фамилия в третьей колонке на листе 'База данных креаторов'
    } else if (targetSheet.getName() === 'Контент-завод') {
      targetSurnameCol = 2; // Фамилия во второй колонке на листе 'Контент-завод'
    }
    if (data[i][targetSurnameCol - 1] === surname) {
      targetSheet.getRange(i + 1, targetCol).setValue(status);
    }
  }
}

//=====Функция для объекта с начальной и конечной датой для запроса Подменные артикулы=====//
function getFormattedDatesForPodmenAPI() {

  // Получаем значения из ячеек C1 и D1
  const startDateStr = sheetPodmen.getRange("C1").getDisplayValue();
  const endDateStr = sheetPodmen.getRange("D1").getDisplayValue();

  // Функция для преобразования формата
  const formatDate = (dateStr) => {
    if (!dateStr) return null;

    // Разбиваем строку на компоненты
    const [day, month, year] = dateStr.split('.');

    // Создаем объект Date (месяцы в JS начинаются с 0)
    const date = new Date(year, month - 1, day);

    // Проверяем валидность даты
    if (isNaN(date.getTime())) {
      throw new Error(`Неверный формат даты: ${dateStr}`);
    }

    // Форматируем в YYYY-MM-DD
    return Utilities.formatDate(date, Session.getScriptTimeZone(), "yyyy-MM-dd");
  };

  return {
    start: formatDate(startDateStr),
    end: formatDate(endDateStr)
  };
}

//=====Функция для массива артикулов для запроса Подменные артикулы=====//
//Получает массив артикулов из колонки C (начиная с C2), игнорируя пустые ячейки //
// @returns {number[]} Массив числовых артикулов //

function getNmIdsFromSheetPodmen() {
  // Получаем все значения колонки C, начиная со строки 2
  const columnData = sheetPodmen.getRange("C2:C" + sheetPodmen.getLastRow()).getDisplayValues();

  // Фильтруем пустые ячейки и преобразуем в числа
  return columnData
    .flat() // Преобразуем двумерный массив в одномерный
    .filter(cell => cell !== "") // Игнорируем пустые ячейки
    .map(Number) // Преобразуем строки в числа
    .filter(id => !isNaN(id)); // Удаляем возможные некорректные значения
}

// Получает артикулы из листа во внешней таблице
function getNmIdsFromExternalSheet(spreadsheet, sheetName) {
  try {
    const sheet = spreadsheet.getSheetByName(sheetName);
    if (!sheet) return [];

    return sheet.getRange("A2:A" + sheet.getLastRow())
      .getValues()
      .flat()
      .filter(cell => cell !== "")
      .map(Number)
      .filter(id => !isNaN(id));
  } catch (e) {
    console.error(`Ошибка при чтении таблицы ${spreadsheet.getName()}:`, e);
    return [];
  }
}

    "method": "POST",

function transformSearchData(originalArray) {
  return originalArray.map(function (item) {
    return {
      text: item.text,
      nmId: item.nmId,
      frequency: item.frequency.current, // Количество обращений с поисковым запросом
      weekFrequency: item.weekFrequency, // Количество обращений с поисковым запросом за неделю
      medianPosition: item.medianPosition.current, // Медианная позиция в поиске
      avgPosition: item.avgPosition.current, // Средняя позиция в поиске
      openCard: item.openCard.current, // Переходы в карточку товара
      addToCart: item.addToCart.current, // Добавления в корзину
      openToCart: item.openToCart.current, // Конверсия в корзину (добавления / переходы)
      orders: item.orders.current, // Заказы из поиска
      cartToOrder: item.cartToOrder.current, // Конверсия в заказ (заказы / добавления в корзину)
      visibility: item.visibility.current // Видимость товара (%)
    };
  });
}

// Сопоставляет данные из массива с таблицей и записывает результаты
// @param {Array} dataArray - Массив объектов с данными

function matchAndWriteData(dataArray) {
  Logger.log("Начинаем запись результатов")
  // Получаем данные из колонок C (артикулы) и F (подменные артикулы)
  const lastRow = sheetPodmen.getLastRow();
  const nmIds = sheetPodmen.getRange("C3:C" + lastRow).getValues().flat();
  const substituteTexts = sheetPodmen.getRange("F3:F" + lastRow).getValues().flat();

  // Подготовим массив для записи результатов
  const results = [];

  // Перебираем строки таблицы
  for (let i = 0; i < nmIds.length; i++) {
    const nmId = nmIds[i];
    const text = substituteTexts[i];
    // Ищем совпадения в массиве данных
    const matchedData = dataArray.find(item =>
      item.nmId == nmId && item.text == text
    );

    // Формируем строку результатов
    if (matchedData) {
      results.push([
        matchedData.frequency || 0,
        matchedData.openCard || 0,
        matchedData.addToCart || 0,
        matchedData.orders || 0
      ]);
    } else {
      // Если совпадений нет - пустые значения
      results.push(["", "", "", ""]);
    }
  }
  // Записываем результаты в колонки J-M начиная с 3 строки
  if (results.length > 0) {
    sheetPodmen.getRange(3, 10, results.length, 4).setValues(results);
  }
  Logger.log("Результаты записаны")
}

}

// ====== ВСПОМОГАТЕЛЬНАЯ ФУНКЦИЯ: РАЗБИВКА МАССИВА НА ЧАСТИ ====== //
function chunkArray(array, chunkSize) {
  const chunks = [];
  for (let i = 0; i < array.length; i += chunkSize) {
    chunks.push(array.slice(i, i + chunkSize));
  }
  return chunks;
}

// ====== ФУНКЦИЯ: ПОЛУЧЕНИЕ ВСЕХ nmIds ДЛЯ КАБИНЕТА ====== //
function getAllNmIdsByCabinet(cabinetName) {
  if (!sheetNmIds) {
    console.error("❌ Лист nmIds не инициализирован");
    return [];
  }
  
  const lastRow = sheetNmIds.getLastRow();
  if (lastRow < 2) {
    console.log("⚠️ На листе nmIds нет данных (нужна хотя бы строка с заголовком)");
    return [];
  }
  
  // Получаем все данные, начиная со строки 2 (после заголовка)
  const dataRange = sheetNmIds.getRange(2, 1, lastRow - 1, 2);
  const data = dataRange.getValues();
  
  const nmIds = [];
  const seenIds = new Set();
  
  for (let i = 0; i < data.length; i++) {
    const nmId = data[i][0];      // Столбец A - nmId
    const cabinet = data[i][1];   // Столбец B - кабинет
    
    // Пропускаем пустые nmId
    if (!nmId || nmId.toString().trim() === "") {
      continue;
    }
    
    // Пропускаем, если кабинет не соответствует нужному
    if (!cabinet || cabinet.toString().trim() !== cabinetName) {
      continue;
    }
    
    const nmIdNum = Number(nmId);
    if (!isNaN(nmIdNum) && nmIdNum > 0 && !seenIds.has(nmIdNum)) {
      nmIds.push(nmIdNum);
      seenIds.add(nmIdNum);
    }
  }
  
  console.log(`✅ Для кабинета "${cabinetName}" найдено ВСЕГО ${nmIds.length} nmId`);
  return nmIds;
}

// ====== ФУНКЦИЯ: ЗАГРУЗКА МАППИНГА ПОДМЕННЫХ АРТИКУЛОВ ====== //
function _poiskovie_zaprosi_loadPodmenMapping() {
  console.log("Загружаем маппинг подменных артикулов из листа 'Аналитика по запросам'...");
  
  if (!sheetZaprosy) {
    console.error("❌ Лист не инициализирован");
    return new Map();
  }
  
  const lastRow = sheetZaprosy.getLastRow();
  if (lastRow < 3) {
    console.log("❌ На листе нет данных (требуются данные с строки 3)");
    return new Map();
  }
  
  const dataRange = sheetZaprosy.getRange(3, 1, lastRow - 2, 2);
  const data = dataRange.getValues();
  
  const mapping = new Map();
  let loadedCount = 0;
  let duplicateCount = 0;
  
  data.forEach((row) => {
    const searchWord = String(row[0]).trim();
    const mainArtikul = row[1];
    
    if (!searchWord || searchWord === "") {
      return;
    }
    
    if (mapping.has(searchWord)) {
      duplicateCount++;
      const currentValue = mapping.get(searchWord);
      
      if (!Array.isArray(currentValue)) {
        mapping.set(searchWord, [currentValue, mainArtikul]);
      } else {
        currentValue.push(mainArtikul);
      }
    } else {
      mapping.set(searchWord, mainArtikul);
    }
    
    loadedCount++;
  });
  
  console.log(`✅ Загружено ${loadedCount} соответствий для фильтрации переходов`);
  if (duplicateCount > 0) {
    console.log(`⚠️ Найдено ${duplicateCount} дублирующихся запросов (будут обработаны как массивы)`);
  }
  
  return mapping;
}

// ====== ФУНКЦИЯ: ПОЛУЧЕНИЕ ОСНОВНОГО АРТИКУЛА ====== //
function _poiskovie_zaprosi_shouldCountTransitions(searchWord, nmId, mapping) {
  const nmIdStr = String(nmId);
  
  if (mapping.has(searchWord)) {
    const mainArtikulValue = mapping.get(searchWord);
    
    if (Array.isArray(mainArtikulValue)) {
      return mainArtikulValue.some(main => {
        if (!main && main !== 0) return false;
        return String(main) === nmIdStr;
      });
    } else {
      if (!mainArtikulValue && mainArtikulValue !== 0) {
        return true;
      }
      return String(mainArtikulValue) === nmIdStr;
    }
  }
  
  return true;
}

// ====== ФУНКЦИЯ ПОЛУЧЕНИЯ ДАТ ====== //
function getFormattedDatesForPodmenAPI(mode = 'sheet', manualStartDate = null, manualEndDate = null) {
  let startDateStr, endDateStr;

  switch (mode) {
    case 'sheet':
      startDateStr = sheetZaprosy.getRange("A1").getDisplayValue();
      endDateStr = sheetZaprosy.getRange("B1").getDisplayValue();
      break;
    case 'manual':
      startDateStr = manualStartDate;
      endDateStr = manualEndDate;
      break;
    case 'auto':
      const dates = getLastWeekDates();
      startDateStr = dates.start;
      endDateStr = dates.end;
      break;
    default:
      throw new Error("Неизвестный режим получения дат");
  }

  const formatDate = (dateStr) => {
    if (!dateStr) return null;
    const [day, month, year] = dateStr.split('.');
    const date = new Date(year, month - 1, day);
    if (isNaN(date.getTime())) {
      throw new Error(`Неверный формат даты: ${dateStr}`);
    }
    return Utilities.formatDate(date, Session.getScriptTimeZone(), "yyyy-MM-dd");
  };

  return {
    start: formatDate(startDateStr),
    end: formatDate(endDateStr),
    displayStart: startDateStr,
    displayEnd: endDateStr
  };
}

function getLastWeekDates() {
  const today = new Date();
  const dayOfWeek = today.getDay();
  const lastMonday = new Date(today);
  lastMonday.setDate(today.getDate() - dayOfWeek - 6);
  const lastSunday = new Date(lastMonday);
  lastSunday.setDate(lastMonday.getDate() + 6);

  const formatToDDMMYYYY = (date) => {
    const day = String(date.getDate()).padStart(2, '0');
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const year = date.getFullYear();
    return `${day}.${month}.${year}`;
  };

  return {
    start: formatToDDMMYYYY(lastMonday),
    end: formatToDDMMYYYY(lastSunday)
  };
}

function getSearchWordsFromSheetZaprosy() {
  if (!sheetZaprosy) {
    console.log("❌ Лист не инициализирован");
    return { words: [], rowMap: {}, lastRow: 0 };
  }

  const lastRow = sheetZaprosy.getLastRow();
  if (lastRow < 3) {
    console.log("❌ На листе нет данных (нужны данные с строки 3)");
    return { words: [], rowMap: {}, lastRow: 0 };
  }

  const columnData = sheetZaprosy.getRange("A3:A" + lastRow).getValues();
  const words = [];
  const rowMap = {};

  for (let i = 0; i < columnData.length; i++) {
    const cellValue = columnData[i][0];
    if (cellValue && cellValue.toString().trim() !== "") {
      const word = cellValue.toString().trim();
      words.push(word);
      rowMap[word] = i + 3;
    }
  }

  console.log(`✅ Получено ${words.length} запросов с листа`);
  return { words, rowMap, lastRow };
}

function findFirstEmptyColumn() {
  const lastColumn = sheetZaprosy.getLastColumn();
  return lastColumn + 1;
}

function createHeadersAndDates(startColumn, startEndDates) {

  sheetZaprosy.getRange(2, startColumn, 1, 4).setValues([datesRow]);

}

function getNextBlockColor(startColumn) {
  if (startColumn === 1) return "#FED6BC";
  
  try {
    const lastCellColor = sheetZaprosy.getRange(1, startColumn - 1).getBackground();
    return (lastCellColor === "#fed6bc" || lastCellColor === "#FED6BC") ? "#DEF7FE" : "#FED6BC";
  } catch (error) {
    return "#FED6BC";
  }
}

// ====== ОСНОВНАЯ ФУНКЦИЯ: АНАЛИЗ ДЛЯ КАБИНЕТА С ПАКЕТНОЙ ОБРАБОТКОЙ ====== //
function analyzeSearchWordsForCabinet(apiKey, dates, searchWordsData, cabinetName, limit) {
  console.log(`\n📊 НАЧАЛО АНАЛИЗА ДЛЯ КАБИНЕТА: ${cabinetName}`);
  console.log(`📋 Лимит запроса (количество поисковых слов в ответе): ${limit}`);
  
  // Получаем ВСЕ nmIds для кабинета
  const allNmIds = getAllNmIdsByCabinet(cabinetName);
  
  if (allNmIds.length === 0) {
    console.log(`❌ Нет nmIds для кабинета "${cabinetName}"`);
    return {};
  }
  
  console.log(`📦 Всего nmId для ${cabinetName}: ${allNmIds.length}`);
  
  // Разбиваем на порции по 50 (ограничение API)
  const nmIdChunks = chunkArray(allNmIds, 50);
  console.log(`📦 Разбито на ${nmIdChunks.length} порций по 50 nmId (последняя порция: ${nmIdChunks[nmIdChunks.length-1].length} nmId)`);
  
  // Хранилище для объединённых результатов по всем порциям
  let aggregatedResults = {};
  let totalRequests = 0;
  let successfulRequests = 0;
  
  // Обрабатываем каждую порцию
  for (let i = 0; i < nmIdChunks.length; i++) {
    const chunk = nmIdChunks[i];
    console.log(`\n--- Пакет ${i+1}/${nmIdChunks.length} для ${cabinetName} (${chunk.length} nmId) ---`);
    console.log(`📝 nmId в этом пакете: ${chunk.slice(0, 5).join(", ")}${chunk.length > 5 ? "..." : ""}`);
    
    try {
      // Создаем запрос для текущей порции nmId
      const options = createSearchWordsRequestOptions(apiKey, dates, limit, chunk);
      const response = sendSearchWordsRequest(options, `${cabinetName} (пакет ${i+1})`);
      
      totalRequests++;
      
      if (response && response.length > 0) {
        successfulRequests++;
        console.log(`✅ Пакет ${i+1}: получено ${response.length} записей`);
        
        // Анализируем текущую порцию и суммируем с предыдущими
        const chunkResults = analyzeResultsChunk(response, searchWordsData, cabinetName, _poiskovie_zaprosi_podmenMapping);
        
        // Суммируем результаты с общими
        for (const word in chunkResults) {
          if (!aggregatedResults[word]) {
            aggregatedResults[word] = {
              totalFrequency: 0,
              totalOrders: 0,
              totalOpenCard: 0,
              totalAddToCart: 0,
              matchCount: 0,
              includedTransitionsCount: 0,
              excludedTransitionsCount: 0
            };
          }
          
          aggregatedResults[word].totalFrequency += chunkResults[word].totalFrequency;
          aggregatedResults[word].totalOrders += chunkResults[word].totalOrders;
          aggregatedResults[word].totalOpenCard += chunkResults[word].totalOpenCard;
          aggregatedResults[word].totalAddToCart += chunkResults[word].totalAddToCart;
          aggregatedResults[word].matchCount += chunkResults[word].matchCount;
          aggregatedResults[word].includedTransitionsCount += chunkResults[word].includedTransitionsCount;
          aggregatedResults[word].excludedTransitionsCount += chunkResults[word].excludedTransitionsCount;
        }
      } else {
        console.log(`⚠️ Пакет ${i+1}: нет данных или ошибка`);
      }
      
      // ====== ВАЖНО: ПАУЗА МЕЖДУ ПАКЕТАМИ ====== //
      // API Wildberries: 3 запроса в минуту = 1 запрос каждые 20 секунд
      // Делаем паузу 21 секунду для безопасности
      if (i < nmIdChunks.length - 1) {
        const waitTime = 21000; // 21 секунда
        console.log(`⏳ Ожидание ${waitTime/1000} секунд перед следующим пакетом (соблюдение лимитов API)...`);
        Utilities.sleep(waitTime);
      }
      
    } catch (error) {
      console.log(`❌ Ошибка при обработке пакета ${i+1} для ${cabinetName}:`, error.toString());
      
      // При ошибке тоже делаем паузу перед следующим пакетом
      if (i < nmIdChunks.length - 1) {
        const waitTime = 30000; // 30 секунд при ошибке
        console.log(`⏳ Ошибка! Ожидание ${waitTime/1000} секунд перед следующим пакетом...`);
        Utilities.sleep(waitTime);
      }
    }
  }
  
  console.log(`\n📊 ИТОГО ДЛЯ ${cabinetName}:`);
  console.log(`✅ Успешных запросов: ${successfulRequests}/${totalRequests}`);
  console.log(`📊 Найдено уникальных поисковых слов: ${Object.keys(aggregatedResults).length}`);
  
  // Выводим детальную статистику по каждому слову (ограничим вывод, чтобы не забивать лог)
  let wordCount = 0;
  for (const word in aggregatedResults) {
    if (wordCount >= 20) {
      console.log(`... и еще ${Object.keys(aggregatedResults).length - 20} слов`);
      break;
    }
    const data = aggregatedResults[word];
    console.log(`\n🎯 Слово: "${word}"`);
    console.log(`📊 Найдено совпадений: ${data.matchCount}`);
    console.log(`📈 Итоговые суммы: Частота=${data.totalFrequency}, Переходы=${data.totalOpenCard} (учтено: ${data.includedTransitionsCount}, исключено: ${data.excludedTransitionsCount}), Добавления=${data.totalAddToCart}, Заказы=${data.totalOrders}`);
    
    if (data.excludedTransitionsCount > 0) {
      console.log(`⚠️ Переходы отфильтрованы: исключено ${data.excludedTransitionsCount} записей (не основной артикул)`);
    }
    wordCount++;
  }
  
  return aggregatedResults;
}

// ====== НОВАЯ ФУНКЦИЯ: АНАЛИЗ ОДНОГО ПАКЕТА ДАННЫХ ====== //
function analyzeResultsChunk(data, searchWordsData, cabinetName, podmenMapping) {
  if (data.length === 0) {
    return {};
  }

  const wordsAnalysis = {};
  const { words } = searchWordsData;
  
  words.forEach(word => {
    const wordLower = word.toLowerCase();
    const matches = [];

    data.forEach(item => {
      if (item.text && item.text.toLowerCase().includes(wordLower)) {
        matches.push({
          text: item.text,
          frequency: item.frequency,
          openCard: item.openCard,
          addToCart: item.addToCart,
          orders: item.orders,
          nmId: item.nmId
        });
      }
    });

    if (matches.length > 0) {
      const totals = matches.reduce((acc, match) => {
        acc.frequency += match.frequency;
        acc.addToCart += match.addToCart;
        acc.orders += match.orders;
        
        const shouldCountTransitions = _poiskovie_zaprosi_shouldCountTransitions(
          word,
          match.nmId,
          podmenMapping
        );
        
        if (shouldCountTransitions) {
          acc.openCard += match.openCard;
          acc.includedTransitionsCount++;
        } else {
          acc.excludedTransitionsCount++;
        }
        
        return acc;
      }, { 
        frequency: 0, 
        openCard: 0, 
        addToCart: 0, 
        orders: 0,
        includedTransitionsCount: 0,
        excludedTransitionsCount: 0
      });

      wordsAnalysis[word] = {
        totalFrequency: totals.frequency,
        totalOrders: totals.orders,
        totalOpenCard: totals.openCard,
        totalAddToCart: totals.addToCart,
        matchCount: matches.length,
        includedTransitionsCount: totals.includedTransitionsCount,
        excludedTransitionsCount: totals.excludedTransitionsCount
      };
    }
  });

  return wordsAnalysis;
}

// ====== СОЗДАНИЕ ЗАПРОСА К API (с передачей конкретной порции nmIds) ====== //
function createSearchWordsRequestOptions(apiKey, currentPeriod, limit, nmIdsChunk) {
  if (!nmIdsChunk || nmIdsChunk.length === 0) {
    throw new Error(`❌ Нет nmIds для запроса`);
  }
  
  const formData = {
    "currentPeriod": {
      "start": currentPeriod.start,
      "end": currentPeriod.end
    },
    "nmIds": nmIdsChunk,
    "topOrderBy": "openCard",
    "includeSubstitutedSKUs": true,
    "includeSearchTexts": true,
    "orderBy": {
      "field": "visibility",
      "mode": "asc"
    },
    "limit": limit
  };

  return {
    "method": "POST",
    "muteHttpExceptions": true,
    "headers": {
      "Authorization": apiKey,
      "Content-Type": "application/json"
    },
    "payload": JSON.stringify(formData)
  };
}

// ====== ОТПРАВКА ЗАПРОСА ====== //
function sendSearchWordsRequest(options, requestName) {
  const wbApiURL = "https://seller-analytics-api.wildberries.ru/api/v2/search-report/product/search-texts";

  try {
    console.log(`📤 Отправляем запрос ${requestName}...`);
    const response = UrlFetchApp.fetch(wbApiURL, options);
    const responseCode = response.getResponseCode();

    if (responseCode !== 200) {
      console.log(`❌ Ошибка API для ${requestName}:`, response.getContentText());
      return [];
    }

    const data = JSON.parse(response.getContentText());

    if (!data?.data?.items) {
      console.log(`⚠️ Нет данных в ответе для ${requestName}`);
      return [];
    }

    console.log(`✅ Получено элементов для ${requestName}: ${data.data.items.length}`);
    return transformSearchData(data.data.items);

  } catch (error) {
    console.log(`❌ Ошибка при обработке ответа для ${requestName}:`, error.toString());
    return [];
  }
}

// ====== ПРЕОБРАЗОВАНИЕ ДАННЫХ ====== //
function transformSearchData(originalArray) {
  if (!Array.isArray(originalArray)) return [];

  return originalArray.map(function (item) {
    const frequency = typeof item.frequency === 'object' ?
      (item.frequency.current || item.frequency || 0) :
      (item.frequency || 0);

    const openCard = typeof item.openCard === 'object' ?
      (item.openCard.current || item.openCard || 0) :
      (item.openCard || 0);

    const addToCart = typeof item.addToCart === 'object' ?
      (item.addToCart.current || item.addToCart || 0) :
      (item.addToCart || 0);

    const orders = typeof item.orders === 'object' ?
      (item.orders.current || item.orders || 0) :
      (item.orders || 0);

    return {
      text: item.text || "",
      nmId: item.nmId || 0,
      frequency: Number(frequency) || 0,
      openCard: Number(openCard) || 0,
      addToCart: Number(addToCart) || 0,
      orders: Number(orders) || 0
    };
  }).filter(item => item.text !== "");
}

// ====== ОБЪЕДИНЕНИЕ РЕЗУЛЬТАТОВ ООО И ИП ====== //
function combineResults(oooResults, ipResults, words) {
  const combinedResults = {};
  
  words.forEach(word => {
    const oooData = oooResults[word] || {
      totalFrequency: 0,
      totalOrders: 0,
      totalOpenCard: 0,
      totalAddToCart: 0
    };
    
    const ipData = ipResults[word] || {
      totalFrequency: 0,
      totalOrders: 0,
      totalOpenCard: 0,
      totalAddToCart: 0
    };
    
    combinedResults[word] = {
      totalFrequency: oooData.totalFrequency + ipData.totalFrequency,
      totalOrders: oooData.totalOrders + ipData.totalOrders,
      totalOpenCard: oooData.totalOpenCard + ipData.totalOpenCard,
      totalAddToCart: oooData.totalAddToCart + ipData.totalAddToCart
    };
  });
  
  console.log(`\n📊 ОБЪЕДИНЕНИЕ РЕЗУЛЬТАТОВ:`);
  console.log(`✅ Результаты ООО и ИП объединены (данные за один период суммированы по одинаковым словам)`);
  
  return combinedResults;
}

// ====== ОСНОВНАЯ ЛОГИКА ====== //
function executeDataSaving(currentPeriod, pastPeriod) {
  console.log("=== СБОР ДАННЫХ ЗА ТЕКУЩИЙ ПЕРИОД ===");
  console.log("📅 Текущий период:", currentPeriod.displayStart, "-", currentPeriod.displayEnd);
  console.log("📅 Предыдущий период:", pastPeriod.displayStart, "-", pastPeriod.displayEnd);

  const allNmIds = getUniqueNmIdsFromListSecondScript();
  console.log(`📦 Всего уникальных артикулов: ${allNmIds.length}`);
  
  let allResults = [];

  if (allNmIds.length === 0) {
    console.log("🔄 Режим: Запрос без артикулов (все товары)");
    const results = analyzeForDataSaving(apiKeyOOO, currentPeriod, pastPeriod, LIMIT_OOO, []);
    if (results && results.length > 0) {
      allResults = results;
      console.log(`✅ Получено ${results.length} записей без артикулов`);
    } else {
      console.log(`❌ Нет данных при запросе без артикулов`);
      return;
    }
  } else {
    console.log("🔄 Режим: Запрос с конкретными артикулами");
    const batches = [];
    for (let i = 0; i < allNmIds.length; i += BATCH_SIZE) {
      batches.push(allNmIds.slice(i, i + BATCH_SIZE));
    }
    
    console.log(`🔄 Разбито на ${batches.length} групп по ${BATCH_SIZE} артикулов`);
    let batchNumber = 1;
    let hasData = false;
    
    for (const batch of batches) {
      console.log(`\n📦 === ГРУППА ${batchNumber}/${batches.length} ===`);
      console.log(`📡 Запрашиваем данные для ${batch.length} артикулов`);
      
      const results = analyzeForDataSaving(apiKeyOOO, currentPeriod, pastPeriod, LIMIT_OOO, batch);
      
      if (results && results.length > 0) {
        console.log(`✅ Получено ${results.length} записей`);
        allResults.push(...results);
        hasData = true;
      } else {
        console.log(`❌ Нет данных для этой группы`);
      }
      
      batchNumber++;
      
      if (batchNumber <= batches.length) {
        console.log(`⏳ Пауза ${BATCH_DELAY/1000} секунд...`);
        Utilities.sleep(BATCH_DELAY);
      }
    }
    
    if (!hasData) {
      console.log("❌ Нет данных ни для одной группы артикулов");
      return;
    }
  }

  if (allResults.length > 0) {
    console.log(`\n✅ Данные успешно получены (${allResults.length} записей). Очищаем лист...`);
    clearOldData();
    writeResultsToCSVSheet(allResults, currentPeriod, pastPeriod);
    console.log(`\n🎉 ВСЕГО сохранено: ${allResults.length} записей`);
  } else {
    console.log("\n❌ Нет данных для сохранения");
  }
}

// ====== АНАЛИЗ ДЛЯ СОХРАНЕНИЯ ДАННЫХ ====== //
function analyzeForDataSaving(apiKey, currentPeriod, pastPeriod, limit, nmIds) {
  console.log(`Запрашиваем данные для ${nmIds.length} артикулов`);

  try {
    const options = createDataRequestOptions(apiKey, currentPeriod, pastPeriod, limit, nmIds);
    
    console.log(`⏳ Пауза ${REQUEST_DELAY/1000} сек перед запросом...`);
    Utilities.sleep(REQUEST_DELAY);
    
    const response = sendDataRequest(options);

    if (response && response.length > 0) {
      return response;
    } else {
      return [];
    }

  } catch (error) {
    console.log(`❌ Ошибка при запросе:`, error.toString());
    return [];
  }
}

// ====== ЗАПРОС К API С ДВУМЯ ПЕРИОДАМИ ====== //
function createDataRequestOptions(apiKey, currentPeriod, pastPeriod, limit, nmIds) {
  const formData = {
    "currentPeriod": {
      "start": currentPeriod.start,
      "end": currentPeriod.end
    },
    "pastPeriod": {
      "start": pastPeriod.start,
      "end": pastPeriod.end
    },
    "nmIds": nmIds,
    "topOrderBy": "openCard",
    "includeSubstitutedSKUs": true,
    "includeSearchTexts": true,
    "orderBy": {
      "field": "openCard",
      "mode": "desc"
    },
    "limit": limit
  };

  return {
    "method": "POST",
    "muteHttpExceptions": true,
    "headers": {
      "Authorization": apiKey,
      "Content-Type": "application/json"
    },
    "payload": JSON.stringify(formData)
  };
}

function sendDataRequest(options) {
  const wbApiURL = "https://seller-analytics-api.wildberries.ru/api/v2/search-report/product/search-texts";

  try {
    const response = UrlFetchApp.fetch(wbApiURL, options);
    const responseCode = response.getResponseCode();

    if (responseCode === 429) {
      console.log("⚠️ Превышен лимит запросов. Увеличиваем паузу...");
      Utilities.sleep(60000);
      return [];
    } else if (responseCode !== 200) {
      console.log(`❌ Ошибка API (${responseCode}):`, response.getContentText());
      return [];
    }

    const data = JSON.parse(response.getContentText());

    if (!data?.data?.items) {
      console.log(`ℹ️ Нет данных в ответе`);
      return [];
    }

    console.log(`📊 Получено ${data.data.items.length} записей от API`);
    return transformDataForCSVSheetAdvanced(data.data.items);

  } catch (error) {
    console.log(`❌ Критическая ошибка при обработке ответа:`, error.toString());
    return [];
  }
}

// ====== АЛЬТЕРНАТИВНАЯ ТРАНСФОРМАЦИЯ ДАННЫХ ====== //
function transformDataForCSVSheetAdvanced(originalArray) {
  if (!Array.isArray(originalArray)) return [];

  console.log(`\n🔍 АЛЬТЕРНАТИВНАЯ ТРАНСФОРМАЦИЯ ДАННЫХ:`);
  console.log("Количество элементов:", originalArray.length);

  const transformField = (field, fieldName = '') => {
    if (typeof field === 'object' && field !== null) {
      const currentValue = Number(field.current) || 0;
      const dynamicsValue = Number(field.dynamics) || 0;
      
      console.log(`📊 ${fieldName}: current=${currentValue}, dynamics=${dynamicsValue}`);
      
      // АНАЛИЗ ТИПА DYNAMICS
      let displayValue = 0;
      
      if (fieldName === 'frequency') {
        // Для frequency: dynamics = абсолютное изменение (не проценты)
        displayValue = dynamicsValue; // Абсолютное число
        console.log(`🔢 ${fieldName}: АБСОЛЮТНОЕ изменение: ${dynamicsValue}`);
      } else {
        // Для остальных полей: dynamics = проценты
        displayValue = dynamicsValue / 100; // Конвертируем в десятичную дробь
        console.log(`📈 ${fieldName}: ПРОЦЕНТНОЕ изменение: ${dynamicsValue}%`);
      }
      
      return {
        current: currentValue,
        dynamics: displayValue
      };
    }
    
    console.log(`⚠️ ${fieldName}: Нет данных`);
    return {
      current: 0,
      dynamics: 0
    };
  };

  return originalArray.map(function (item, index) {
    const frequencyData = transformField(item.frequency, 'frequency');
    const openCardData = transformField(item.openCard, 'openCard');
    const addToCartData = transformField(item.addToCart, 'addToCart');
    const ordersData = transformField(item.orders, 'orders');

    // Логируем первые 2 записи
    if (index < 2) {
      console.log(`\n📋 ТРАНСФОРМАЦИЯ "${item.text}":`);
      console.log("Запросы:", frequencyData.current, `(abs: ${frequencyData.dynamics})`);
      console.log("Переходы:", openCardData.current, `(%: ${(openCardData.dynamics * 100).toFixed(1)}%)`);
      console.log("Корзина:", addToCartData.current, `(%: ${(addToCartData.dynamics * 100).toFixed(1)}%)`);
      console.log("Заказы:", ordersData.current, `(%: ${(ordersData.dynamics * 100).toFixed(1)}%)`);
    }

    return {
      vendorCode: item.vendorCode || "",
      nmId: item.nmId || 0,
      name: item.name || "",
      text: item.text || "",
      frequency: frequencyData.current,
      openCard: openCardData.current,
      addToCart: addToCartData.current,
      orders: ordersData.current,
      frequencyDynamics: frequencyData.dynamics,
      openCardDynamics: openCardData.dynamics,
      addToCartDynamics: addToCartData.dynamics,
      ordersDynamics: ordersData.dynamics
    };
  });
}

// ====== УСЛОВНОЕ ФОРМАТИРОВАНИЕ ====== //
function applyConditionalFormattingAdvanced(rowCount) {
  if (rowCount === 0) return;
  
  try {
    const sheet = sheetCSVExport;
    sheet.clearConditionalFormatRules();
    
    let newRules = [];
    
    // Для абсолютных изменений (колонка 6) - запросы
    const absRange = sheet.getRange(4, 6, rowCount, 1);
    const absGreenRule = SpreadsheetApp.newConditionalFormatRule()
      .setRanges([absRange])
      .whenNumberGreaterThan(0)
      .setBackground("#C6EFCE")
      .setFontColor("#006100")
      .build();
    
    const absRedRule = SpreadsheetApp.newConditionalFormatRule()
      .setRanges([absRange])
      .whenNumberLessThan(0)
      .setBackground("#FFC7CE")
      .setFontColor("#9C0006")
      .build();
    
    const absGrayRule = SpreadsheetApp.newConditionalFormatRule()
      .setRanges([absRange])
      .whenNumberEqualTo(0)
      .setBackground("#F3F3F3")
      .setFontColor("#666666")
      .build();
    
    newRules.push(absGreenRule, absRedRule, absGrayRule);
    
    // Для процентных изменений (колонки 8, 10, 12)
    const percentageColumns = [8, 10, 12];
    percentageColumns.forEach(col => {
      const range = sheet.getRange(4, col, rowCount, 1);
      
      const greenRule = SpreadsheetApp.newConditionalFormatRule()
        .setRanges([range])
        .whenNumberGreaterThan(0)
        .setBackground("#C6EFCE")
        .setFontColor("#006100")
        .build();
      
      const redRule = SpreadsheetApp.newConditionalFormatRule()
        .setRanges([range])
        .whenNumberLessThan(0)
        .setBackground("#FFC7CE")
        .setFontColor("#9C0006")
        .build();
      
      const grayRule = SpreadsheetApp.newConditionalFormatRule()
        .setRanges([range])
        .whenNumberEqualTo(0)
        .setBackground("#F3F3F3")
        .setFontColor("#666666")
        .build();
      
      newRules.push(greenRule, redRule, grayRule);
    });
    
    sheet.setConditionalFormatRules(newRules);
    console.log("🎨 Условное форматирование применено (разные типы)");
    
  } catch (error) {
    console.log("⚠️ Не удалось применить условное форматирование:", error.message);
  }
}

// ====== ФУНКЦИИ ДЛЯ РАБОТЫ С ДАТАМИ ====== //
function getFormattedDatesForCSVExport(mode = 'sheet', manualStartDate = null, manualEndDate = null) {
  let startDateStr, endDateStr;

  switch (mode) {
    case 'sheet':
      startDateStr = sheetCSVExport.getRange("A2").getDisplayValue();
      endDateStr = sheetCSVExport.getRange("B2").getDisplayValue();
      
      if (!startDateStr || !endDateStr || startDateStr.trim() === "" || endDateStr.trim() === "") {
        console.log("⚠️ Даты в таблице пустые. Используем автоматический режим.");
        const dates = getLastWeekDatesCSV();
        startDateStr = dates.start;
        endDateStr = dates.end;
      }
      break;
    case 'manual':
      startDateStr = manualStartDate;
      endDateStr = manualEndDate;
      break;
    case 'auto':
      const dates = getLastWeekDatesCSV();
      startDateStr = dates.start;
      endDateStr = dates.end;
      break;
    default:
      throw new Error("Неизвестный режим получения дат");
  }

  const formatDate = (dateStr) => {
    if (!dateStr) return null;
    
    try {
      const [day, month, year] = dateStr.split('.');
      if (!day || !month || !year) {
        throw new Error(`Неполная дата: ${dateStr}`);
      }
      
      const date = new Date(year, month - 1, day);
      if (isNaN(date.getTime())) {
        throw new Error(`Неверный формат даты: ${dateStr}`);
      }
      return Utilities.formatDate(date, Session.getScriptTimeZone(), "yyyy-MM-dd");
    } catch (error) {
      console.log(`❌ Ошибка при форматировании даты "${dateStr}":`, error.message);
      return null;
    }
  };

  const currentStart = formatDate(startDateStr);
  const currentEnd = formatDate(endDateStr);

  if (!currentStart || !currentEnd) {
    console.log("❌ Не удалось получить корректные даты. Используем прошлую неделю.");
    const fallbackDates = getLastWeekDatesCSV();
    const currentPeriod = {
      start: formatDate(fallbackDates.start),
      end: formatDate(fallbackDates.end),
      displayStart: fallbackDates.start,
      displayEnd: fallbackDates.end
    };
    
    const pastPeriod = calculatePastPeriod(currentPeriod.start, currentPeriod.end);
    return { currentPeriod, pastPeriod };
  }

  const currentPeriod = {
    start: currentStart,
    end: currentEnd,
    displayStart: startDateStr,
    displayEnd: endDateStr
  };

  const pastPeriod = calculatePastPeriod(currentPeriod.start, currentPeriod.end);

  return {
    currentPeriod,
    pastPeriod
  };
}

function calculatePastPeriod(currentStart, currentEnd) {
  const startDate = new Date(currentStart);
  const endDate = new Date(currentEnd);
  
  const durationMs = endDate - startDate;
  const durationDays = Math.round(durationMs / (1000 * 60 * 60 * 24)) + 1;
  
  console.log(`📅 Расчет прошлого периода: ${durationDays} дней`);
  
  const pastStart = new Date(startDate);
  pastStart.setDate(pastStart.getDate() - durationDays);
  
  const pastEnd = new Date(endDate);
  pastEnd.setDate(pastEnd.getDate() - durationDays);
  
  const formatToDDMMYYYY = (date) => {
    const day = String(date.getDate()).padStart(2, '0');
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const year = date.getFullYear();
    return `${day}.${month}.${year}`;
  };

  const result = {
    start: Utilities.formatDate(pastStart, Session.getScriptTimeZone(), "yyyy-MM-dd"),
    end: Utilities.formatDate(pastEnd, Session.getScriptTimeZone(), "yyyy-MM-dd"),
    displayStart: formatToDDMMYYYY(pastStart),
    displayEnd: formatToDDMMYYYY(pastEnd)
  };
  
  console.log(`📅 Прошлый период: ${result.displayStart} - ${result.displayEnd}`);
  
  return result;
}

function getLastWeekDatesCSV() {
  const today = new Date();
  const dayOfWeek = today.getDay();
  const lastMonday = new Date(today);
  lastMonday.setDate(today.getDate() - dayOfWeek - 6);
  const lastSunday = new Date(lastMonday);
  lastSunday.setDate(lastMonday.getDate() + 6);

  const formatToDDMMYYYY = (date) => {
    const day = String(date.getDate()).padStart(2, '0');
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const year = date.getFullYear();
    return `${day}.${month}.${year}`;
  };

  return {
    start: formatToDDMMYYYY(lastMonday),
    end: formatToDDMMYYYY(lastSunday)
  };
}
